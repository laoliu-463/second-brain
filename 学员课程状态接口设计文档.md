## 五、枚举定义（eams-domain）

```java
package com.zeroone.star.project.j5.studentstatus.enums;

import java.util.Arrays;
import java.util.Map;
import java.util.stream.Collectors;

public enum StudentLessonStatusEnum {
    NOT_STARTED("未开始"),
    IN_PROGRESS("进行中"),
    COMPLETED("已完成"),
    SUSPENDED("已暂停");

    private final String desc;

    StudentLessonStatusEnum(String desc) {
        this.desc = desc;
    }

    public String getDesc() {
        return desc;
    }

    public static Map<String, String> toMap() {
        return Arrays.stream(values())
                .collect(Collectors.toMap(
                        StudentLessonStatusEnum::name,
                        StudentLessonStatusEnum::getDesc
                ));
    }

    public static String getDescByName(String name) {
        try {
            return StudentLessonStatusEnum.valueOf(name).getDesc();
        } catch (IllegalArgumentException e) {
            return null;
        }
    }
}
```

## 六、Controller 层实现（eams-j5-studentstatus）

```java
package com.zeroone.star.j5.studentstatus.controller;

import com.zeroone.star.project.dto.PageDTO;
import com.zeroone.star.project.dto.j5.studentstatus.*;
import com.zeroone.star.project.j5.studentstatus.StudentLessonStatusApis;
import com.zeroone.star.j5.studentstatus.service.IStudentLessonStatusService;
import com.zeroone.star.project.query.j5.studentstatus.StudentLessonStatusQuery;
import com.zeroone.star.project.vo.JsonVO;
import com.zeroone.star.project.vo.j5.studentstatus.StudentLessonStatusVO;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.*;

import javax.annotation.Resource;

@RestController
@RequestMapping("j5/student-lesson-status")
@Api(tags = "学员课程状态管理")
public class StudentLessonStatusController implements StudentLessonStatusApis {

    @Resource
    private IStudentLessonStatusService studentLessonStatusService;

    @GetMapping("list")
    @ApiOperation("分页查询")
    @Override
    public JsonVO<PageDTO<StudentLessonStatusVO>> list(StudentLessonStatusQuery query) {
        PageDTO<StudentLessonStatusVO> result = studentLessonStatusService.list(query);
        return JsonVO.success(result);
    }

    @PostMapping("batch-set-status")
    @ApiOperation("批量设置课程状态")
    @Override
    public JsonVO<Integer> batchSetStatus(@Validated @RequestBody BatchSetStatusDTO dto) {
        int count = studentLessonStatusService.batchSetStatus(dto);
        return JsonVO.success(count);
    }

    @PostMapping("toggle-lesson-count")
    @ApiOperation("批量停课/复课")
    @Override
    public JsonVO<Integer> batchToggleLessonStatus(@Validated @RequestBody BatchToggleLessonStatusDTO dto) {
        int count = studentLessonStatusService.batchToggleLessonStatus(dto);
        return JsonVO.success(count);
    }

    @PostMapping("restore")
    @ApiOperation("批量还原课程进度")
    @Override
    public JsonVO<Integer> batchRestore(@Validated @RequestBody BatchRestoreDTO dto) {
        int count = studentLessonStatusService.batchRestore(dto);
        return JsonVO.success(count);
    }
}
```
